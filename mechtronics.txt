'{$STAMP BS2}
'{$PBASIC 2.5}


rand VAR NIB
error_num VAR NIB
button_pushes VAR NIB

RANDOM rand
error_num = rand // 8

DEBUG HOME "The error number is :", DEC error_num, CR

SELECT error_num
	CASE 0
		DEBUG "No error, yay!", CR
	CASE 1
		HIGH 1
	CASE 2
		HIGH 3
	CASE 3
		HIGH 1
		HIGH 2
	CASE 4
		HIGH 3
	CASE 5
		HIGH 1
		HIGH 3
	CASE 6
		HIGH 2
		HIGH 3
	CASE 7
		HIGH 1
		HIGH 2
		HIGH 3
ENDSELECT

DO
	IF IN15 = 1 THEN
		button_pushes = button_pushes + 1
		DO
		LOOP UNTIL IN15 = 0
	ENDIF
LOOP UNTIL button_pushes = error_num

LOW 1
LOW 3
LOW 5

DEBUG "Error is clear!", CR
